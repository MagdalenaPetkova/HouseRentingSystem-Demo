pipeline {
  agent any
  stages {
    stage('Restore') {
      steps {
        bat "dotnet restore HouseRentingSystem.sln"
      }
    }
    stage('Build') {
      steps {
        bat "dotnet build HouseRentingSysten.sln --configuration Release"
      }
    }
    stage('Test') {
      steps {
        bat "dotnet test HouseRentingSysten.Tests/HouseRentingSysten.Tests.csproj"
      }
    }
  }
}
